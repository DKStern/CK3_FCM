#Сделать красивым
fcm_become_beautiful = {
	category = interaction_category_cheat_menu
	use_diplomatic_range = no
	common_interaction = yes
	
	is_shown = {
		NOT = {
			scope:recipient = {
				has_trait = beauty_good_3
			}
		}
	}

	on_accept = {
		scope:recipient = {
			add_trait = beauty_good_3
		}
	}

	auto_accept = yes

	ai_will_do = 0
}
