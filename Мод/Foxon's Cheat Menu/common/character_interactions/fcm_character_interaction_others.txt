#Убить
fcm_kill = {
	category = fcm_interaction_category
	common_interaction = yes

	is_shown = {
		NOT = { scope:recipient = scope:actor }
	}

	on_accept = {
		scope:recipient = {		
			death = natural 
		}
	}
	auto_accept = yes

	ai_will_do = 0
}

#Свадьба
fcm_marry = {
	category = fcm_interaction_category
	common_interaction = yes

	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:recipient = {
			AND = { 
				is_married = no
				is_adult = yes
				gender = scope:actor
			}
		}
	}

	on_accept = {
		if = {
			limit = { 
				scope:actor = { gender = female }
			}
			scope:actor = {
				marry_matrilineal = scope:recipient
			}		
		}
		else = {
			scope:actor = {		
				marry = scope:recipient
			}
		}
	}
	auto_accept = yes

	ai_will_do = 0
}

#Забеременеть
fcm_impregnate ={
	category = fcm_interaction_category
	common_interaction = yes

	is_shown = {
		NOT = { scope:actor = scope:recipient }
		OR = {
			AND = {
				scope:actor = {
					is_male = yes
					age >= 16
				}
				scope:recipient = {
					is_female = yes
					is_pregnant = no
					age >= 16
				}
			}
			AND = {
				scope:recipient = {
					is_male = yes
					age >= 16
				}
				scope:actor = {
					is_female = yes
					is_pregnant = no
					age >= 16
				}
			}
		}
	}

	on_accept = {
		if = {
			limit = {
				scope:actor = {
					is_male = yes
				}
			}
			scope:recipient = {
				make_pregnant = {
					father = scope:actor
				}
			}
		}
		else = {
			scope:actor = {
				make_pregnant = {
					father = scope:recipient
				}
			}
		}
	}

	auto_accept = yes
	ai_will_do = 0
}

	category = fcm_interaction_category
	common_interaction = yes

	is_shown = {
		NOT = { scope:recipient = scope:actor }
	}

	on_accept = {
		scope:actor = {
			imprison = {
				target = scope:recipient
				type = dungeon
			}
		}
	}
	auto_accept = yes

	ai_will_do = 0
}
